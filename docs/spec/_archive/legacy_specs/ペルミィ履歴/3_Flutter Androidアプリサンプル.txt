import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

/// アプリのルート
class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      home: CounterPage(),
    );
  }
}

/// カウンター画面
class CounterPage extends StatefulWidget {
  const CounterPage({super.key});

  @override
  State<CounterPage> createState() => _CounterPageState();
}

class _CounterPageState extends State<CounterPage> {
  int _count = 0;

  void _increment() {
    setState(() {
      _count++;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Flutter Sample'),
      ),
      body: Center(
        child: Text(
          '$_count',
          style: const TextStyle(fontSize: 40),
        ),
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: _increment,
        child: const Icon(Icons.add),
      ),
    );
  }
}

import 'dart:async';

void main() => runApp(const MyApp());

class MyApp extends StatefulWidget {
  const MyApp({super.key});
  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  StreamSubscription<List<SharedMediaFile>>? _sub;
  String _inputPreview = '';
  List<String> _filePaths = [];

  @override
  void initState() {
    super.initState();

    // アプリ起動中に共有されたとき
    _sub = ReceiveSharingIntent.instance.getMediaStream().listen((files) async {
      await _handleIncoming(files);
    });

    // 共有からアプリが起動されたとき（コールドスタート）
    ReceiveSharingIntent.instance.getInitialMedia().then((files) async {
      if (files.isNotEmpty) await _handleIncoming(files);
    });
  }

  Future<void> _handleIncoming(List<SharedMediaFile> files) async {
    // 共有が「テキスト」だけのケースは別経路になることがあるので、
    // まずはファイル共有（トーク履歴ファイル）を想定して受ける。
    final paths = files.map((f) => f.path).where((p) => p.isNotEmpty).toList();

    // テキスト共有の場合は getTextStream / getInitialText を使う
    // （後述）
    setState(() {
      _filePaths = paths;
      _inputPreview = paths.isEmpty ? '(no files)' : paths.take(3).join('\n');
    });
  }

  @override
  void dispose() {
    _sub?.cancel();
    ReceiveSharingIntent.instance.reset();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: const Text('Talk Import')),
        body: Padding(
          padding: const EdgeInsets.all(12),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.stretch,
            children: [
              const Text('受信プレビュー:'),
              const SizedBox(height: 8),
              Text(_inputPreview),
              const SizedBox(height: 16),
              ElevatedButton(
                onPressed: _filePaths.isEmpty ? null : () async {
                  final text = await _readFirstFileAsText(_filePaths.first);
                  setState(() => _inputPreview = text.substring(0, text.length.clamp(0, 2000)));
                  // TODO: ここで解析→AI呼び出し
                },
                child: const Text('先頭ファイルを読み込む'),
              ),
            ],
          ),
        ),
      ),
    );
  }

  Future<String> _readFirstFileAsText(String path) async {
    final f = File(path);
    // LINEのエクスポートがzipやtxt等の場合があるので、実運用は拡張子で分岐
    return await f.readAsString();
  }
}

class HomePage extends StatefulWidget {
  const HomePage({super.key});

class _HomePageState extends State<HomePage> {
  StreamSubscription<List<SharedMediaFile>>? _mediaSub;
  String _sharedText = ''; // 共有で受け取ったテキスト（整形済み）
  String _manualText = ''; // 手入力・コピペ用

  @override
  void initState() {
    super.initState();

    // アプリ起動中に共有されたとき
    _mediaSub = ReceiveSharingIntent.instance
        .getMediaStream()
        .listen((List<SharedMediaFile> media) {
      setState(() {
        _sharedText = _extractText(media);
      });
    }, onError: (err) {
      debugPrint('getMediaStream error: $err');
    });

    // 共有からアプリが起動されたとき（コールドスタート）
    ReceiveSharingIntent.instance
        .getInitialMedia()
        .then((List<SharedMediaFile> media) {
      if (media.isNotEmpty) {
        setState(() {
          _sharedText = _extractText(media);
        });
      }
    });
  }

  @override

    // コールバック再利用防止
    ReceiveSharingIntent.instance.reset();
    super.dispose();

  /// SharedMediaFile のリストからテキストを組み立てる
  String _extractText(List<SharedMediaFile> media) {

    final buf = StringBuffer();
    for (final m in media) {
      // テキスト or text/plain 的なものは path に本文が入る
      final mime = m.mimeType ?? '';
      if (mime.startsWith('text')) {
        buf.writeln(m.path);
      } else {
        // 画像・動画などは一旦パスだけ（将来ここを拡張）
        buf.writeln('[${mime.isEmpty ? m.type : mime}] ${m.path}');
      }
    }
    return buf.toString().trim();
  }

        },
        body: jsonEncode(<String, dynamic>{
          'text': text,
        }),
      );

      if (resp.statusCode != 200) {
        throw Exception('statusCode: ${resp.statusCode}');
      }

      final json = jsonDecode(resp.body) as Map<String, dynamic>;

      final summary = (json['summary'] ?? '') as String;
      final repliesDynamic = json['replies'];
      final replies = repliesDynamic is List
          ? repliesDynamic.map((e) => e.toString()).toList()
          : <String>[];

      setState(() {
        _summary = summary;
        _replies = replies;
      });
    } catch (e) {
      setState(() {
        _error = 'AI呼び出しに失敗しました: $e';
      });
    } finally {
      if (mounted) {
        setState(() {
          _loading = false;
        });
      }
    }
  }

app = FastAPI()


class TalkRequest(BaseModel):
    text: str

    summary: str
    replies: List[str]


@app.post("/api/talk/assist", response_model=TalkResponse)
async def talk_assist(req: TalkRequest) -> TalkResponse:
    text = req.text.strip()

    # ---- 今はダミー実装 ----
    no_newline = text.replace("\n", " ")
    max_len = 80
    if len(no_newline) <= max_len:
        summary = f"【ダミー要約】{no_newline}"
    else:
        summary = f"【ダミー要約】{no_newline[:max_len]}…"

    short = text if len(text) <= 30 else text[:30] + "…"
    replies = [
        "了解です、こちらの内容で進めておきます。",
        "ありがとうございます。少し確認したい点があるので、あとでお時間もらってもいいですか？",
        f"メッセージ拝見しました。「{short}」の部分について、もう少し具体的に教えてもらえると助かります。",
    ]
    # -----------------------

    return TalkResponse(summary=summary, replies=replies)

@app.post("/api/talk/assist", response_model=TalkResponse)
async def talk_assist(req: TalkRequest) -> TalkResponse:
    text = req.text.strip()

    if not text:
        return TalkResponse(summary="（入力が空です）", replies=[])

    # プロンプト設計：
    # - まずトークの要約
    # - 次に返信案を3パターン（丁寧 / カジュアル / 超短文）
    system_prompt = """あなたはLINEなどのチャットの返信を考えるアシスタントです。
以下の制約を守ってください。
- 相手との関係性やトーンをできるだけ読み取る
- 必要以上に盛らず、自然な日本語にする
- 自動送信前提ではないので、最終判断はユーザーがする前提で提案だけ行う
- 出力フォーマットを厳守する
出力フォーマット:
1行目: 要約（50〜120文字程度）
2行目以降: 返信案を3つ。各行は「- 」から始める。"""

    user_prompt = f"""以下はLINEのトーク履歴です。内容を理解したうえで、
- 会話内容の要約
- 返信候補を3パターン（丁寧 / カジュアル / 超短文）

を出力フォーマットどおりに作成してください。

--- トーク履歴 ---
{text}
"""

    try:
        completion = client.responses.create(
            model="gpt-4.1-mini",
            input=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt},
            ],
        )

        content = completion.output[0].content[0].text.strip()

        # 期待されるフォーマット:
        # 1行目: 要約
        # 2行目以降: "- ..." の返信案
        lines = [line.strip() for line in content.splitlines() if line.strip()]
        if not lines:
            return TalkResponse(summary="（AIからの応答が空です）", replies=[])

        summary = lines[0]
        replies: list[str] = []
        for line in lines[1:]:
            if line.startswith("-"):
                replies.append(line.lstrip("- ").strip())
            else:
                # 念のため、行頭が "-" でなくても返信候補とみなす
                replies.append(line)

        # 念のため3つまでに制限
        replies = replies[:3]

        return TalkResponse(summary=summary, replies=replies)

    except Exception as e:
        # ログだけ出して、クライアントには汎用エラーを返す
        print(f"[ERROR] talk_assist OpenAI error: {e}")
        return TalkResponse(
            summary="（AI呼び出しに失敗しました。しばらく時間をおいて再度お試しください。）",
            replies=[],
        )

from typing import List
import os

from fastapi import FastAPI
from pydantic import BaseModel
from openai import OpenAI

app = FastAPI()

# OpenAIクライアント（環境変数 OPENAI_API_KEY を使用）
client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))


class TalkRequest(BaseModel):
    text: str
    tone: str = "standard"  # 'standard' | 'night' | 'business'


class TalkResponse(BaseModel):
    summary: str
    replies: List[str]


@app.post("/api/talk/assist", response_model=TalkResponse)
async def talk_assist(req: TalkRequest) -> TalkResponse:
    text = req.text.strip()
    tone = (req.tone or "standard").lower()

    if not text:
        return TalkResponse(summary="（入力が空です）", replies=[])

    # トーン別の説明
    if tone == "night":
        tone_desc = (
            "夜職（接客業）の営業LINEを想定したトーンで返信案を作成してください。"
            "距離感はフレンドリーだが軽すぎない、敬語とタメ口を柔軟に混ぜて構いません。"
            "売上や来店につながるが、押しつけがましくならない提案を心がけてください。"
        )
    elif tone == "business":
        tone_desc = (
            "ビジネスチャット（上司・顧客・取引先）を想定した丁寧な敬語で返信案を作成してください。"
            "クッション言葉を適度に入れ、感情的にならず、簡潔かつ具体的に伝えてください。"
        )
    else:
        tone_desc = (
            "一般的な友人・知人とのやりとりを想定した自然な日本語で返信案を作成してください。"
            "丁寧すぎず、くだけすぎないバランスをとってください。"
        )

    system_prompt = f"""あなたはLINEなどのチャットの返信を考えるアシスタントです。
以下の制約を守ってください。
- 相手との関係性やトーンをできるだけ読み取る
- 必要以上に盛らず、自然な日本語にする
- 自動送信前提ではないので、最終判断はユーザーがする前提で「提案」だけ行う
- 出力フォーマットを厳守する
- トーンの条件: {tone_desc}

出力フォーマット:
1行目: 要約（50〜120文字程度）
2行目以降: 返信案を3つ。各行は「- 」から始める。"""

    user_prompt = f"""以下はLINEのトーク履歴です。内容を理解したうえで、
- 会話内容の要約
- 返信候補を3パターン（丁寧 / カジュアル / 超短文）

を上記の出力フォーマットどおりに作成してください。

--- トーク履歴 ---
{text}
"""

    try:
        completion = client.responses.create(
            model="gpt-4.1-mini",
            input=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt},
            ],
        )

        # Responses API の結果からテキストを取り出す
        content = completion.output[0].content[0].text.strip()

        lines = [line.strip() for line in content.splitlines() if line.strip()]
        if not lines:
            return TalkResponse(summary="（AIからの応答が空でした）", replies=[])

        # 1行目: 要約
        summary = lines[0]

        # 2行目以降: 返信候補（- から始まる行を優先）
        replies: List[str] = []
        for line in lines[1:]:
            if line.startswith("-"):
                replies.append(line.lstrip("-").strip())
            else:
                # 念のため、フォーマット崩れも候補として拾う
                replies.append(line)

        # 最大3件に制限
        replies = replies[:3]

        return TalkResponse(summary=summary, replies=replies)

    except Exception as e:
        # サーバ側ログ
        print(f"[ERROR] talk_assist OpenAI error: {e}")
        # クライアントには汎用メッセージで返す
        return TalkResponse(
            summary="（AI呼び出しに失敗しました。しばらく時間をおいて再度お試しください。）",
            replies=[],
        )
