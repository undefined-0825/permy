def write_log(record: dict) -> None:
    """エラーになっても処理を止めない、超簡易ログ"""
    try:
        record = {**record}
        record["ts"] = datetime.utcnow().isoformat()
        with LOG_FILE.open("a", encoding="utf-8") as f:
            f.write(json.dumps(record, ensure_ascii=False) + "\n")
    except Exception:
        # ログ失敗は黙殺
        pass

def preprocess_text(raw_text: str, tone: ToneLiteral) -> Tuple[str, str, int, int]:
    """
    長文対策の前処理。
    戻り値: (used_text, mode, original_len, used_len)
      mode: "full" | "trimmed" | "summarized"
    """
    text = raw_text.strip()
    original_len = len(text)

    # 生テキストが短ければそのまま
    if original_len <= MAX_USED_CHARS:
        return text, "full", original_len, original_len

    # まずは頭 + 尻でトリミング
    head = text[: MAX_USED_CHARS // 2]
    tail = text[-MAX_USED_CHARS // 2 :]
    trimmed = head + "\n...\n" + tail

    if original_len <= MAX_RAW_CHARS:
        used_len = len(trimmed)
        return trimmed, "trimmed", original_len, used_len

    # さらに長い場合は要約フェーズ（summarized）
    try:
        summary = summarize_conversation(text, tone)
        used_text = summary
        mode = "summarized"
    except Exception as e:
        # 要約に失敗したらトリム版で妥協
        print(f"[WARN] summarize_conversation failed: {e}")
        used_text = trimmed
        mode = "trimmed-fallback"

    used_len = len(used_text)
    return used_text, mode, original_len, used_len

def parse_ai_output(content: str) -> Tuple[str, List[str]]:
    """
    モデルから返ってきたテキストを「要約」「返信案」に分解する。
    - '要約：' / '要約:' 行は見出しとして扱う
    - '返信案：' / '返信案:' 行は見出しとしてスキップ
    - '- ' や '・' で始まる行を返信候補として優先
    """
    lines = [line.strip() for line in content.splitlines() if line.strip()]
    if not lines:
        return "（AIからの応答が空でした）", []

    summary: Optional[str] = None
    summary_index: int = -1

    # 要約行を探す
    for idx, line in enumerate(lines):
        # "要約：" だけの行（中身なし）は見出し扱いでスキップ
        if re.fullmatch(r"要約[:：]?", line):
            continue

        # "要約: XXX" 形式なら、コロン以降を要約として取る
        m = re.match(r"要約[:：]\s*(.+)", line)
        if m:
            summary = m.group(1).strip()
            summary_index = idx
            break

        # それ以外は最初に出てきた非空行を要約とみなす
        summary = line
        summary_index = idx
        break

    if summary is None:
        summary = "（要約を抽出できませんでした）"
        summary_index = -1

    replies: List[str] = []

    # 返信候補を探す
    start_idx = summary_index + 1
    for line in lines[start_idx:]:
        # "返信案" 見出しっぽい行はスキップ
        if re.fullmatch(r"返信案[:：]?", line):
            continue

        # "- " や "・" で始まる行を優先
        if line.startswith("-"):
            replies.append(line.lstrip("-").strip())
            continue
        if line.startswith("・"):
            replies.append(line.lstrip("・").strip())
            continue

        # "1. xxx" / "１）xxx" など番号付きも拾う
        if re.match(r"^[0-9０-９]+[.)．）]\s*", line):
            line2 = re.sub(r"^[0-9０-９]+[.)．）]\s*", "", line)
            replies.append(line2.strip())
            continue

        # それ以外: 既に返信候補があれば無視、まだ無ければ拾っておく
        if not replies:
            replies.append(line)

    # 最大3件
    replies = replies[:3]

    return summary, replies


def parse_night_output(content: str) -> List[str]:
    """
    夜職専用：要約なし前提で返信3案だけを抜き出す。
    1. / 2. / 3. 形式 or 箇条書きを拾う。
    """
    lines = [line.strip() for line in content.splitlines() if line.strip()]
    if not lines:
        return []

    replies: List[str] = []

    for line in lines:
        # 番号付き
        if re.match(r"^[0-9０-９]+[.)．）]\s*", line):
            line2 = re.sub(r"^[0-9０-９]+[.)．）]\s*", "", line)
            replies.append(line2.strip())
            continue

        # 箇条書きも一応拾う
        if line.startswith("-"):
            replies.append(line.lstrip("-").strip())
            continue
        if line.startswith("・"):
            replies.append(line.lstrip("・").strip())
            continue

    return replies[:3]

def _hash_token(token: str) -> str:
    return hashlib.sha256(token.encode("utf-8")).hexdigest()

def _utc_iso(dt: datetime) -> str:
    return dt.astimezone(timezone.utc).isoformat()

    return datetime.now(timezone.utc)

def ensure_user(user_id: str) -> None:
    con = connect()
    cur = con.execute("SELECT user_id FROM users WHERE user_id=?", (user_id,))
    if cur.fetchone() is None:
        con.execute(
            "INSERT INTO users(user_id, plan, admin_free, created_at) VALUES(?,?,?,?)",
            (user_id, "free", 0, _utc_iso(_now_utc())),
        )
        # 初期settings（空でもOK。固定NG等はサーバ側で扱う）
        empty = "{}"
        con.execute(

            (user_id, empty, _utc_iso(_now_utc()), empty),
        )
    con.commit()
    con.close()

def issue_tokens(user_id: str) -> tuple[str, str]:
    ensure_user(user_id)

    access = secrets.token_urlsafe(32)
    refresh = secrets.token_urlsafe(48)

    access_hash = _hash_token(access)
    refresh_hash = _hash_token(refresh)

    now = _now_utc()
    access_exp = now + timedelta(seconds=settings.access_ttl_seconds)
    refresh_exp = now + timedelta(seconds=settings.refresh_ttl_seconds)

    con = connect()
    con.execute(
        "INSERT INTO tokens(token_hash,user_id,kind,expires_at,revoked_at,rotated_from,created_at) VALUES(?,?,?,?,?,?,?)",
        (access_hash, user_id, "access", _utc_iso(access_exp), None, None, _utc_iso(now)),
    )
    # 端末1台前提：refreshは常に1本。既存refreshを失効
    con.execute(
        "UPDATE tokens SET revoked_at=? WHERE user_id=? AND kind='refresh' AND revoked_at IS NULL",
        (_utc_iso(now), user_id),
    )
    con.execute(
        "INSERT INTO tokens(token_hash,user_id,kind,expires_at,revoked_at,rotated_from,created_at) VALUES(?,?,?,?,?,?,?)",
        (refresh_hash, user_id, "refresh", _utc_iso(refresh_exp), None, None, _utc_iso(now)),
    )
    con.commit()
    con.close()
    return access, refresh

def rotate_refresh(old_refresh: str) -> tuple[str, str] | None:
    old_hash = _hash_token(old_refresh)
    now = _now_utc()

    con = connect()
    row = con.execute(
        "SELECT user_id, expires_at, revoked_at FROM tokens WHERE token_hash=? AND kind='refresh'",
        (old_hash,),
    ).fetchone()
    if row is None:
        con.close()
        return None

    if row["revoked_at"] is not None:
        con.close()
        return None

    if datetime.fromisoformat(row["expires_at"]) <= now:
        con.close()
        return None

    user_id = row["user_id"]

    # old refresh revoke + new refresh issue
    con.execute("UPDATE tokens SET revoked_at=? WHERE token_hash=?", (_utc_iso(now), old_hash))

    access = secrets.token_urlsafe(32)
    refresh = secrets.token_urlsafe(48)

    access_hash = _hash_token(access)
    refresh_hash = _hash_token(refresh)

    access_exp = now + timedelta(seconds=settings.access_ttl_seconds)
    refresh_exp = now + timedelta(seconds=settings.refresh_ttl_seconds)

    con.execute(
        "INSERT INTO tokens(token_hash,user_id,kind,expires_at,revoked_at,rotated_from,created_at) VALUES(?,?,?,?,?,?,?)",
        (access_hash, user_id, "access", _utc_iso(access_exp), None, None, _utc_iso(now)),
    )
    con.execute(
        "INSERT INTO tokens(token_hash,user_id,kind,expires_at,revoked_at,rotated_from,created_at) VALUES(?,?,?,?,?,?,?)",
        (refresh_hash, user_id, "refresh", _utc_iso(refresh_exp), None, old_hash, _utc_iso(now)),
    )
    con.commit()
    con.close()
    return access, refresh

def revoke_all_refresh(user_id: str) -> None:
    now = _utc_iso(_now_utc())
    con = connect()
    con.execute(
        "UPDATE tokens SET revoked_at=? WHERE user_id=? AND kind='refresh' AND revoked_at IS NULL",
        (now, user_id),
    )
    con.commit()
    con.close()

def verify_access(access: str) -> str | None:
    token_hash = _hash_token(access)
    now = _now_utc()

    con = connect()

    if row["revoked_at"] is not None:

def _system_prompt(tone: Tone, purpose: Purpose | None, is_short_ok: bool) -> str:
    # 返信は常に3案：攻め/標準/控えめ
    # 出力はJSON固定にしてパース事故を消す（最短で安定）
    base = (

        "【最重要】出力は必ずJSONのみ。余計な文章は一切禁止。\n"
        '出力形式: {"replies":["...","...","..."]}\n'
        "repliesは必ず3件。空文字禁止。\n"
        "差分: 1=攻め/踏み込み, 2=標準, 3=控えめ。\n"
        "原則短文禁止。ただし短文許可の場合は短くても良い。\n"
    )
    tone_desc = ""
    if tone == "business":
        tone_desc = "口調: ビジネス。丁寧・簡潔・絵文字なし。\n"
    elif tone == "night":
        tone_desc = "口調: 夜職の営業LINE。重くせず自然に誘導。絵文字は各案0〜3。\n"
    else:
        tone_desc = "口調: 友人知人。自然な日本語。絵文字は各案0〜1。\n"

    purpose_desc = ""
    if tone == "night":
        if purpose == "visit":
            purpose_desc = "目的: 来店を具体化（候補日提示→来れそう？）\n"
        elif purpose == "douhan":
            purpose_desc = "目的: ごはん→自然に同伴へ。\n"
        elif purpose == "revive":
            purpose_desc = "目的: 休眠復活（責めない/返信しやすい問い）。\n"
        else:
            purpose_desc = "目的: 夜職営業（目的が不明なら軽い近況→誘いの布石）。\n"

    short_rule = "短文許可: はい\n" if is_short_ok else "短文許可: いいえ\n"
    return base + tone_desc + purpose_desc + short_rule

def _user_prompt(text: str, settings_json: dict) -> str:
    # settingsは本文ではないので含めてOK（ただし過度に長くしない）
    s = json.dumps(settings_json, ensure_ascii=False)

        f"settings={s}\n"
        "---\n"
        f"{text}\n"
        "---\n"
        "JSONのみで回答。\n"
    )

def _parse_json_replies(raw: str) -> list[str]:
    try:
        obj = json.loads(raw)
        replies = obj.get("replies")
        if not isinstance(replies, list) or len(replies) != 3:
            return []
        out = []
        for r in replies:
            if not isinstance(r, str):
                return []
            t = r.strip()
            if not t:
                return []
            out.append(t)
        return out
    except Exception:
        return []

def generate_replies(
    text: str,
    tone: Tone,
    purpose: Purpose | None,
    settings_json: dict,
    is_pro: bool,
) -> tuple[list[str], bool]:
    """
    戻り値: (replies, used_ai)
    used_ai=False の場合はゲート等で固定返信（消費しない想定）
    """
    if not text.strip():
        raise bad_request("入力が空です")

    # NGゲート（本文は保存しない。ここでブロックする）
    if has_ng_topic(text):
        # 3案は返すがAIを使わない（回数消費なし）
        replies = [
            "その話題はちょっと避けたいかも…別の話にしよ？",
            "ごめんね、その件は今は触れないでおきたい🙏 また別の話しよ？",
            "今その話は控えたいな。最近どうしてる？",
        ]
        return replies, False

    # 短文例外判定（ルールベース）
    is_short_ok = short_allowed(text)

    # Pro品質ループ（最大3回 + 10秒）
    deadline = time.monotonic() + (10.0 if is_pro else 0.0)
    attempts = 3 if is_pro else 1

    sys = _system_prompt(tone, purpose, is_short_ok)
    usr = _user_prompt(text, settings_json)

    for i in range(attempts):
        if is_pro and time.monotonic() > deadline:
            break

        raw = call_openai(sys, usr, temperature=0.7, max_output_tokens=600)
        replies = _parse_json_replies(raw)
        if len(replies) == 3:
            return replies, True

    # 失敗（成功条件未達。回数消費なし）
    return [], True

import hashlib
import secrets
from datetime import datetime, timedelta, timezone
from app.core.config import settings
from app.db.conn import connect
from app.core.errors import bad_request, limit_reached
from app.auth.service import issue_tokens, revoke_all_refresh
from app.plans.service import get_limit

_ALPHABET = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"  # I,O,0,1を除外
_GROUP = 4

def _now_utc() -> datetime:
    return datetime.now(timezone.utc)

def _utc_iso(dt: datetime) -> str:
    return dt.astimezone(timezone.utc).isoformat()

def _hash(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8")).hexdigest()

def _gen_code12() -> str:
    # Base32風12文字
    return "".join(secrets.choice(_ALPHABET) for _ in range(12))

def _format_444(code12: str) -> str:
    return f"{code12[:4]}-{code12[4:8]}-{code12[8:12]}"

def issue_migration_code(user_id: str) -> str:
    # Proのみ（admin_free含む）
    if get_limit(user_id) < settings.pro_daily_limit:
        raise limit_reached()

    code12 = _gen_code12()
    code_hash = _hash(code12)
    now = _now_utc()
    exp = now + timedelta(seconds=settings.migration_ttl_seconds)

    con = connect()
    # 既存の未使用コードは無効化（1個だけ運用）
    con.execute(
        "UPDATE migration_codes SET consumed_at=? WHERE user_id=? AND consumed_at IS NULL",
        (_utc_iso(now), user_id),
    )
    con.execute(
        "INSERT INTO migration_codes(code_hash,user_id,expires_at,consumed_at,fail_count,locked_until,created_at) "
        "VALUES(?,?,?,?,?,?,?)",
        (code_hash, user_id, _utc_iso(exp), None, 0, None, _utc_iso(now)),
    )
    con.commit()
    con.close()
    return _format_444(code12)

def consume_migration_code(code_input: str) -> tuple[str, str, str]:
    """
    成功: (user_id, access, refresh)
    失敗: 例外
    """
    code = (code_input or "").replace("-", "").strip().upper()
    if len(code) != 12:
        raise bad_request("移行コードを確認")

    code_hash = _hash(code)
    now = _now_utc()

    con = connect()
    row = con.execute(
        "SELECT user_id, expires_at, consumed_at, fail_count, locked_until FROM migration_codes WHERE code_hash=?",
        (code_hash,),
    ).fetchone()

    if row is None:
        con.close()
        raise bad_request("移行コードを確認")

    # ロック判定
    if row["locked_until"]:
        locked_until = datetime.fromisoformat(row["locked_until"])
        if locked_until > now:
            con.close()
            raise limit_reached()

    # 期限/消費済み判定
    if row["consumed_at"] is not None:
        con.close()
        raise bad_request("移行コードを確認")

    if datetime.fromisoformat(row["expires_at"]) <= now:
        con.close()
        raise bad_request("移行コードを確認")

    user_id = row["user_id"]

    # consume
    con.execute(
        "UPDATE migration_codes SET consumed_at=? WHERE code_hash=?",
        (_utc_iso(now), code_hash),
    )
    con.commit()
    con.close()

    # 旧端末refresh失効（端末1台前提）
    revoke_all_refresh(user_id)
    access, refresh = issue_tokens(user_id)
    return user_id, access, refresh

def record_migration_fail(code_input: str) -> None:
    code = (code_input or "").replace("-", "").strip().upper()
    if len(code) != 12:
        return
    code_hash = _hash(code)
    now = _now_utc()

    con = connect()
    row = con.execute(

        (code_hash,),

        return

    locked_until = None
    if fail_count >= settings.migration_fail_limit:
        locked_until = _utc_iso(now + timedelta(seconds=settings.migration_lock_seconds))

    con.execute(
        "UPDATE migration_codes SET fail_count=?, locked_until=? WHERE code_hash=?",
        (fail_count, locked_until, code_hash),

    con.commit()
    con.close()

def _parse_json_replies(raw: str) -> list[str]:
    import json
    import re

    def try_load(s: str) -> list[str]:
        obj = json.loads(s)
        replies = obj.get("replies")
        if not isinstance(replies, list) or len(replies) != 3:
            return []
        out = []
        for r in replies:
            if not isinstance(r, str):
                return []
            t = r.strip()
            if not t:
                return []
            out.append(t)
        return out

    # まず素直に
    try:
        ok = try_load(raw)
        if ok:
            return ok
    except Exception:
        pass

    # ```json ... ``` を剥がす
    m = re.search(r"\{.*\}", raw, flags=re.DOTALL)
    if not m:
        return []
    candidate = m.group(0)
    try:
        return try_load(candidate)
    except Exception:
        return []

from openai import OpenAI
from app.core.config import settings

_client = OpenAI(api_key=settings.openai_api_key)

def call_openai(system_prompt: str, user_prompt: str, temperature: float, max_output_tokens: int) -> str:
    """
    OpenAI呼び出し（1責務）
    - ChatCompletionsでJSONを安定生成させる
    - 本文ログはしない（呼び出し側で制御）
    """
    # JSON強制（対応していない環境でも壊れないように、systemで拘束しつつfallback可能な形）
    messages = [
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": user_prompt},
    ]

    # 一番互換が高いルート：chat.completions
    completion = _client.chat.completions.create(
        model=settings.openai_model,
        messages=messages,
        temperature=temperature,
        max_tokens=max_output_tokens,
    )

    content = completion.choices[0].message.content
    return (content or "").strip()
