def _parse_json_replies(raw: str) -> list[str]:
    import json
    import re

    def try_load(s: str) -> list[str]:
        obj = json.loads(s)
        replies = obj.get("replies")
        if not isinstance(replies, list) or len(replies) != 3:
            return []
        out = []
        for r in replies:
            if not isinstance(r, str):
                return []
            t = r.strip()
            if not t:
                return []
            out.append(t)
        return out

    # まず素直に
    try:
        ok = try_load(raw)
        if ok:
            return ok
    except Exception:
        pass

    #

        return []
    candidate = m.group(0)

        return []

def generate_replies(
    text: str,
    tone: Tone,
    purpose: Purpose | None,
    settings_json: dict,
    is_pro: bool,
) -> tuple[list[str], bool]:

    戻り値: (replies, used_ai)
    used_ai=False の場合はゲート等で固定返信（消費しない想定）

    if not text.strip():
        raise bad_request("入力が空です")

    deadline = time.monotonic() + (10.0 if is_pro else 0.0)

    """
    OpenAI呼び出し（1責務）

    completion = _client.chat.completions.create(

    )

    return (content or "").strip()
