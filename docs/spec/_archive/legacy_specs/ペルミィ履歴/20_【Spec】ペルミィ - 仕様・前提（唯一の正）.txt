def summarize_conversation(raw_text: str, tone: ToneLiteral) -> str:

        "- 会話の流れと重要なポイントが分かるように、重要な発言のみを時系列で簡潔にまとめてください。\n"
        "- 感情のニュアンス（怒っている・喜んでいる・困っている等）が分かるように含めてください。\n\n"
        "# 出力\n"
        "- 箇条書きや見出しは使わず、2〜6文程度の自然な日本語にしてください。"
    )

{raw_text}

            {"role": "user", "content": user_prompt},
        ],
        max_output_tokens=256,
        temperature=0.4,
    )

    content = completion.output[0].content[0].text.strip()
    return content


def preprocess_text(raw_text: str, tone: ToneLiteral) -> Tuple[str, str, int, int]:
    """
    長文対策の前処理。
    戻り値: (used_text, mode, original_len, used_len)
      mode: "full" | "trimmed" | "summarized"
    """
    text = raw_text.strip()
    original_len = len(text)

    # 生テキストが短ければそのまま
    if original_len <= MAX_USED_CHARS:
        return text, "full", original_len, original_len

    # まずは頭 + 尻でトリミング
    head = text[: MAX_USED_CHARS // 2]
    tail = text[-MAX_USED_CHARS // 2 :]
    trimmed = head + "\n...\n" + tail

    if original_len <= MAX_RAW_CHARS:
        used_len = len(trimmed)
        return trimmed, "trimmed", original_len, used_len

    # さらに長い場合は要約フェーズ（summarized）
    try:
        summary = summarize_conversation(text, tone)
        used_text = summary
        mode = "summarized"
    except Exception as e:
        # 要約に失敗したらトリム版で妥協
        print(f"[WARN] summarize_conversation failed: {e}")
        used_text = trimmed
        mode = "trimmed-fallback"

    used_len = len(used_text)
    return used_text, mode, original_len, used_len

# メインエンドポイント
# ==========

def parse_ai_output(content: str) -> Tuple[str, List[str]]:
    """
    モデルから返ってきたテキストを「要約」「返信案」に分解する。
    - '要約：' / '要約:' 行は見出しとして扱う
    - '返信案：' / '返信案:' 行は見出しとしてスキップ
    - '- ' や '・' で始まる行を返信候補として優先
    """
    lines = [line.strip() for line in content.splitlines() if line.strip()]
    if not lines:
        return "（AIからの応答が空でした）", []

    summary: Optional[str] = None
    summary_index: int = -1

    # 要約行を探す
    for idx, line in enumerate(lines):
        # "要約：" だけの行（中身なし）は見出し扱いでスキップ
        if re.fullmatch(r"要約[:：]?", line):
            continue

        # "要約: XXX" 形式なら、コロン以降を要約として取る
        m = re.match(r"要約[:：]\s*(.+)", line)
        if m:
            summary = m.group(1).strip()
            summary_index = idx
            break

        # それ以外は最初に出てきた非空行を要約とみなす
        summary = line
        summary_index = idx
        break

    if summary is None:
        summary = "（要約を抽出できませんでした）"
        summary_index = -1

    replies: List[str] = []

    # 返信候補を探す
    start_idx = summary_index + 1
    for line in lines[start_idx:]:
        # "返信案" 見出しっぽい行はスキップ
        if re.fullmatch(r"返信案[:：]?", line):
            continue

        # "- " や "・" で始まる行を優先
        if line.startswith("-"):
            replies.append(line.lstrip("-").strip())
            continue
        if line.startswith("・"):
            replies.append(line.lstrip("・").strip())
            continue

        # "1. xxx" / "１）xxx" など番号付きも拾う
        if re.match(r"^[0-9０-９]+[.)．）]\s*", line):
            line2 = re.sub(r"^[0-9０-９]+[.)．）]\s*", "", line)
            replies.append(line2.strip())
            continue

        # それ以外: 既に返信候補があれば無視、まだ無ければ拾っておく
        if not replies:
            replies.append(line)

    # 最大3件
    replies = replies[:3]

    return summary, replies


def parse_night_output(content: str) -> List[str]:
    """
    夜職専用：要約なし前提で返信3案だけを抜き出す。
    1. / 2. / 3. 形式 or 箇条書きを拾う。
    """
    lines = [line.strip() for line in content.splitlines() if line.strip()]
    if not lines:
        return []

    replies: List[str] = []

    for line in lines:
        # 番号付き
        if re.match(r"^[0-9０-９]+[.)．）]\s*", line):
            line2 = re.sub(r"^[0-9０-９]+[.)．）]\s*", "", line)
            replies.append(line2.strip())
            continue

        # 箇条書きも一応拾う
        if line.startswith("-"):
            replies.append(line.lstrip("-").strip())
            continue
        if line.startswith("・"):
            replies.append(line.lstrip("・").strip())
            continue

    return replies[:3]

@app.post("/api/talk/assist", response_model=TalkResponse)
async def talk_assist(req: TalkRequest) -> TalkResponse:
    raw_text = (req.text or "").strip()
    tone_raw = (req.tone or "standard").lower()

    if not raw_text:
        return TalkResponse(summary="（入力が空です）", replies=[])

    # tone を正規化
    if tone_raw not in ("standard", "night", "business"):
        tone: ToneLiteral = "standard"
    else:
        tone = tone_raw  # type: ignore[assignment]

    # 長文前処理
    used_text, mode, original_len, used_len = preprocess_text(raw_text, tone)

    # system prompt 構築
    if tone == "night" and req.night_mode:
        system_prompt = build_night_system_prompt(req.night_mode)
    elif tone == "night" and not req.night_mode:
        # night 指定なのにモード未指定の場合は、暫定で来店確定扱い
        system_prompt = build_night_system_prompt("visit")  # type: ignore[arg-type]
    else:
        system_prompt = build_system_prompt(tone)

    _, temperature = build_tone_desc_and_temp(tone)
    user_prompt = build_user_prompt(used_text, mode, req.user_style)

    try:
        completion = client.responses.create(
            model=MODEL_NAME,
            input=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt},
            ],
            temperature=temperature,
            max_output_tokens=512,
        )

        # Responses API の結果からテキストを取り出す
        content = completion.output[0].content[0].text.strip()

        if tone == "night":
            # 夜職は要約なし・返信のみ
            replies = parse_night_output(content)
            if not replies:
                write_log(
                    {
                        "event": "parsed_night_without_replies",
                        "tone": tone,
                        "night_mode": req.night_mode,
                        "original_len": original_len,
                        "used_len": used_len,
                        "mode": mode,
                        "raw_content_preview": content[:120],
                    }
                )
            write_log(
                {
                    "event": "success_night",
                    "tone": tone,
                    "night_mode": req.night_mode,
                    "original_len": original_len,
                    "used_len": used_len,
                    "mode": mode,
                    "replies_count": len(replies),
                }
            )
            return TalkResponse(summary=None, replies=replies)

        # standard / business
        summary, replies = parse_ai_output(content)

        if not replies:
            # 返信案がうまく取れなかったケースはログに残しておく

    private fun sendRequest() {
        val text = inputText.text.toString().trim()
        if (text.isEmpty()) {
            Toast.makeText(this, "トーク履歴を入力してください", Toast.LENGTH_SHORT).show()
            return
        }

        val tone = getSelectedTone()
        val nightMode = if (tone == "night") getSelectedNightMode() else null
        val userStyle = if (tone == "night") {
            userStyleEditText.text.toString().trim().ifEmpty { null }
        } else {
            null
        }

        // 夜職で nightMode 未選択なら弾く
        if (tone == "night" && nightMode == null) {
            Toast.makeText(this, "営業目的（来店確定/同伴獲得/休眠復活）を選択してください", Toast.LENGTH_SHORT).show()
            return
        }

        statusText.text = "通信中…"
        generateButton.isEnabled = false

        val url = "$baseUrl/api/talk/assist"

        // JSON リクエスト生成
        val json = JSONObject().apply {
            put("text", text)
            put("tone", tone)
            nightMode?.let { put("night_mode", it) }
            userStyle?.let { put("user_style", it) }
        }

        val mediaType = "application/json; charset=utf-8".toMediaType()
        val body = RequestBody.create(mediaType, json.toString())

        val request = Request.Builder()
            .url(url)
            .post(body)
            .build()

        client.newCall(request).enqueue(object : Callback {
            override fun onFailure(call: Call, e: IOException) {
                runOnUiThread {
                    statusText.text = "通信エラー: ${e.message}"
                    generateButton.isEnabled = true
                }
            }

            override fun onResponse(call: Call, response: Response) {
                val bodyString = response.body?.string().orEmpty()
                if (!response.isSuccessful) {
                    runOnUiThread {
                        statusText.text = "エラー: HTTP ${response.code}"
                        generateButton.isEnabled = true
                    }
                    return
                }

                try {
                    val obj = JSONObject(bodyString)
                    val summary = if (obj.isNull("summary")) null else obj.getString("summary")
                    val repliesJson = obj.getJSONArray("replies")

                    val replies = mutableListOf<String>()
                    for (i in 0 until repliesJson.length()) {
                        replies.add(repliesJson.getString(i))
                    }

                    runOnUiThread {
                        statusText.text = "生成完了"

                        if (tone == "night") {
                            // 夜職は要約なし
                            summaryText.text = ""
                            summaryText.visibility = View.GONE
                        } else {
                            summaryText.visibility = View.VISIBLE
                            summaryText.text = summary ?: ""
                        }

                        reply1Button.text = replies.getOrNull(0) ?: "返信案1"
                        reply2Button.text = replies.getOrNull(1) ?: "返信案2"
                        reply3Button.text = replies.getOrNull(2) ?: "返信案3"

                        generateButton.isEnabled = true
                    }
                } catch (e: Exception) {
                    runOnUiThread {
                        statusText.text = "レスポンス解析エラー: ${e.message}"
                        generateButton.isEnabled = true
                    }
                }
            }
        })
    }

    private fun handleShareIntent() {
        val action = intent?.action
        val type = intent?.type

        if (Intent.ACTION_SEND == action && type != null) {

            // 1) 普通のテキスト共有（EXTRA_TEXT）
            if (type.startsWith("text/")) {
                val sharedText = intent.getStringExtra(Intent.EXTRA_TEXT)
                if (!sharedText.isNullOrBlank()) {
                    inputText.setText(sharedText)
                    statusText.text = "LINEから受け取ったトーク履歴を読み込みました。"
                    return
                }
            }

            // 2) ファイル共有（EXTRA_STREAM, 例: .txt）
            val uri: Uri? = intent.getParcelableExtra(Intent.EXTRA_STREAM)
            if (uri != null) {

            summaryText.text = ""
            return

        summaryText.text = ""
        replyButtons.forEach { it.text = "生成中..." }

        val json = JSONObject().apply {
            put("text", text)
            put("tone", getSelectedTone())
        }

        // OkHttp4 推奨の書き方：toMediaType & toRequestBody を使用
        val body = json.toString()
            .toRequestBody("application/json; charset=utf-8".toMediaType())

        val request = Request.Builder()
            .url("$apiBaseUrl/api/talk/assist")
            .post(body)
            .build()

        launch {
            try {
                val response = withContext(Dispatchers.IO) {
                    client.newCall(request).execute()
                }

                response.use { res ->
                    if (!res.isSuccessful) {
                        withContext(Dispatchers.Main) {
                            statusText.text = "APIエラー: ${res.code}"
                            resetReplies()
                        }
                        return@use
                    }

                    val responseBody = res.body?.string().orEmpty()
                    val obj = JSONObject(responseBody)
                    val summary = obj.optString("summary", "")
                    val replies = obj.optJSONArray("replies") ?: JSONArray()

                    withContext(Dispatchers.Main) {
                        summaryText.text = summary
                        replyButtons.forEachIndexed { index, button ->
                            val textReply = replies.optString(index, "")
                            if (textReply.isNotBlank()) {

                    statusText.text = "通信エラーが発生しました: ${e.localizedMessage}"
                    resetReplies()
                }
            }
        }

            statusText.text = "トーク内容が空です。LINEから共有してから実行してください。"
            resetReplies()
            summaryText.text = ""
            return
        }

        val tone = getSelectedTone()
        val nightMode = if (tone == "night") getSelectedNightMode() else null
        val userStyle = if (tone == "night") {
            userStyleEditText.text.toString().trim().ifEmpty { null }
        } else {
            null
        }

        // 夜職で nightMode 未選択なら弾く
        if (tone == "night" && nightMode == null) {
            statusText.text = "営業目的（来店確定 / 同伴獲得 / 休眠復活）を選択してください。"
            return
        }

        statusText.text = "生成中..."
        summaryText.text = ""
        replyButtons.forEach { it.text = "生成中..." }

        // リクエストJSON
        val json = JSONObject().apply {
            put("text", text)
            put("tone", tone)
            nightMode?.let { put("night_mode", it) }
            userStyle?.let { put("user_style", it) }
        }

        // OkHttp4 推奨の書き方：toMediaType & toRequestBody を使用
        val body = json.toString()
            .toRequestBody("application/json; charset=utf-8".toMediaType())

        val request = Request.Builder()
            .url("$apiBaseUrl/api/talk/assist")
            .post(body)
            .build()

        launch {
            try {
                val response = withContext(Dispatchers.IO) {
                    client.newCall(request).execute()
                }

                response.use { res ->
                    if (!res.isSuccessful) {
                        withContext(Dispatchers.Main) {
                            statusText.text = "APIエラー: ${res.code}"
                            resetReplies()
                        }
                        return@use
                    }

                    val responseBody = res.body?.string().orEmpty()
                    val obj = JSONObject(responseBody)
                    // summary は night の場合 null の可能性あり
                    val summary =
                        if (obj.isNull("summary")) null else obj.optString("summary", null)
                    val replies = obj.optJSONArray("replies") ?: JSONArray()

                    withContext(Dispatchers.Main) {
                        statusText.text = "生成完了。返信案をタップしてコピーしてください。"

                        if (tone == "night") {
                            // 夜職は要約なし
                            summaryText.text = ""
                            summaryText.visibility = android.view.View.GONE
                        } else {
                            summaryText.visibility = android.view.View.VISIBLE
                            summaryText.text = summary ?: ""
                        }

                        replyButtons.forEachIndexed { index, button ->
                            val textReply = replies.optString(index, "")
                            if (textReply.isNotBlank()) {
                                button.text = textReply
                            } else {
                                button.text = "返信案${index + 1}（なし）"
                            }
                        }
                    }
                }
            } catch (e: Exception) {
                withContext(Dispatchers.Main) {
                    statusText.text = "通信エラーが発生しました: ${e.localizedMessage}"
                    resetReplies()
                }
            }
        }
    }

    private fun copyToClipboard(text: String) {
        val clipboard = getSystemService(Context.CLIPBOARD_SERVICE) as ClipboardManager
        clipboard.setPrimaryClip(ClipData.newPlainText("reply", text))
    }

        super.onDestroy()

    }

    private fun handleShareIntent() {
        val action = intent?.action
        val type = intent?.type

        if (Intent.ACTION_SEND == action && type != null) {

            // 1) 普通のテキスト共有（EXTRA_TEXT）
            if (type.startsWith("text/")) {
                val sharedText = intent.getStringExtra(Intent.EXTRA_TEXT)
                if (!sharedText.isNullOrBlank()) {
                    inputText.setText(sharedText)
                    statusText.text = "LINEから受け取ったトーク履歴を読み込みました。"
                    return
                }
            }

            val uri: Uri? = intent.getParcelableExtra(Intent.EXTRA_STREAM)
            if (uri != null) {
                try {
                    contentResolver.openInputStream(uri)?.use { input ->
                        val text = input.bufferedReader(Charsets.UTF_8).readText()
                        inputText.setText(text)
                        statusText.text = "LINEのトーク履歴ファイルを読み込みました。"
                    }
                } catch (e: Exception) {
                    e.printStackTrace()
                    statusText.text = "トーク履歴ファイルの読み込みに失敗しました。"
                }
            }
        }

private fun callApi() {
    val text = inputText.text.toString().trim()
    if (text.isBlank()) {
        statusText.text = "トーク内容が空です。LINEから共有してから実行してください。"
        resetReplies()
        summaryText.text = ""
        return
    }

    val tone = getSelectedTone()
    val nightMode = getSelectedNightMode()
    val userStyle = if (tone == "night") {
        userStyleEditText.text.toString().trim().ifEmpty { null }
    } else {
        null
    }

    statusText.text = "ペルミィが返信案を考えています…"
    summaryText.text = ""
    replyContainer.visibility = View.GONE   // 生成前は隠す
    replyButtons.forEach { it.text = "生成中..." }
    showLoading(true)

    val json = JSONObject().apply {
        put("text", text)
        put("tone", tone)
        nightMode?.let { put("night_mode", it) }
        userStyle?.let { put("user_style", it) }
    }

    val body = json.toString()
        .toRequestBody("application/json; charset=utf-8".toMediaType())

    val request = Request.Builder()
        .url("$apiBaseUrl/api/talk/assist")
        .post(body)
        .build()

    launch {
        try {
            val response = withContext(Dispatchers.IO) {
                client.newCall(request).execute()
            }

            response.use { res ->
                if (!res.isSuccessful) {
                    withContext(Dispatchers.Main) {
                        statusText.text = "APIエラー: ${res.code}"
                        resetReplies()
                        showLoading(false)
                    }
                    return@use
                }

                val responseBody = res.body?.string().orEmpty()
                val obj = JSONObject(responseBody)
                val summary =
                    if (obj.isNull("summary")) null else obj.optString("summary", null)
                val replies = obj.optJSONArray("replies") ?: JSONArray()

                withContext(Dispatchers.Main) {
                    statusText.text = "返信案ができました。タップでコピーしてください。"

                    if (tone == "night") {
                        summaryText.text = ""
                        summaryText.visibility = View.GONE
                    } else {
                        summaryText.visibility = View.VISIBLE
                        summaryText.text = summary ?: ""
                    }

                    replyButtons.forEachIndexed { index, button ->
                        val textReply = replies.optString(index, "")
                        if (textReply.isNotBlank()) {
                            button.text = textReply
                        } else {
                            button.text = "返信案${index + 1}（なし）"
                        }
                    }
                    // ここで初めて返信案コンテナを見せる
                    replyContainer.visibility = View.VISIBLE

                    showLoading(false)
                }
            }
        } catch (e: Exception) {
            withContext(Dispatchers.Main) {
                statusText.text = "通信エラーが発生しました: ${e.localizedMessage}"
                resetReplies()
                showLoading(false)
            }
        }
    }
}

                    showLoading(false)

    override fun onCreate(savedInstanceState: Bundle?) {

    private fun setupToneSpinner(initialIndex: Int) {

    private fun saveToneIndex(index: Int) {

    /**
     * LINEなどからの共有インテントを処理
     */
    private fun handleShareIntent() {
        val action = intent?.action
        val type = intent?.type

        if (Intent.ACTION_SEND == action && type != null) {

            // 1) 普通のテキスト共有（EXTRA_TEXT）
            if (type.startsWith("text/")) {
                val sharedText = intent.getStringExtra(Intent.EXTRA_TEXT)
                if (!sharedText.isNullOrBlank()) {
                    inputText.setText(sharedText)
                    statusText.text = "LINEから受け取ったトーク履歴を読み込みました。"
                    return
                }
            }

            // 2) ファイル共有（EXTRA_STREAM, 例: .txt）
            val uri: Uri? = intent.getParcelableExtra(Intent.EXTRA_STREAM)
            if (uri != null) {
                try {
                    contentResolver.openInputStream(uri)?.use { input ->
                        val text = input.bufferedReader(Charsets.UTF_8).readText()
                        inputText.setText(text)
                        statusText.text = "LINEのトーク履歴ファイルを読み込みました。"
                    }
                } catch (e: Exception) {
                    e.printStackTrace()
                    statusText.text = "トーク履歴ファイルの読み込みに失敗しました。"
                }
            }
        }
    }

    /**
     * スピナーで選択されているトーンをAPI用の文字列に変換
     */
    private fun getSelectedTone(): String {
        return when (toneSpinner.selectedItemPosition) {
            1, 2, 3 -> "night"   // 夜職3パターンはすべて night
            4 -> "business"
            else -> "standard"
        }
    }

    private fun getSelectedNightMode(): String? {

            1 -> "visit"   // 夜職（来店確定）
            2 -> "douhan"  // 夜職（同伴獲得）
            3 -> "revive"  // 夜職（休眠復活）
            else -> null
        }
    }

    private fun resetReplies() {
        replyButtons.forEachIndexed { index, button ->
            button.text = "返信案${index + 1}"
        }
        replyContainer.visibility = View.GONE
    }

    private fun showLoading(isLoading: Boolean) {
        if (isLoading) {
            progressBar.visibility = View.VISIBLE
            findViewById<Button>(R.id.generateButton).isEnabled = false
            findViewById<Button>(R.id.clearButton).isEnabled = false
            toneSpinner.isEnabled = false
            inputText.isEnabled = false
            userStyleEditText.isEnabled = false
        } else {
            progressBar.visibility = View.GONE
            findViewById<Button>(R.id.generateButton).isEnabled = true
            findViewById<Button>(R.id.clearButton).isEnabled = true
            toneSpinner.isEnabled = true
            inputText.isEnabled = true
            userStyleEditText.isEnabled = true
        }
    }

    private fun callApi() {
        val text = inputText.text.toString().trim()
        if (text.isBlank()) {
            statusText.text = "トーク内容が空です。LINEから共有してから実行してください。"
            resetReplies()
            summaryText.text = ""
            return
        }

        val tone = getSelectedTone()
        val nightMode = getSelectedNightMode()
        val userStyle = if (tone == "night") {
            userStyleEditText.text.toString().trim().ifEmpty { null }
        } else {
            null
        }

        statusText.text = "ペルミィが返信案を考えています…"
        summaryText.text = ""
        replyContainer.visibility = View.GONE
        replyButtons.forEach { it.text = "生成中..." }
        showLoading(true)

        // リクエストJSON
        val json = JSONObject().apply {
            put("text", text)
            put("tone", tone)
            nightMode?.let { put("night_mode", it) }
            userStyle?.let { put("user_style", it) }
        }

        val body = json.toString()
            .toRequestBody("application/json; charset=utf-8".toMediaType())

        val request = Request.Builder()
            .url("$apiBaseUrl/api/talk/assist")
            .post(body)
            .build()

        launch {
            try {
                val response = withContext(Dispatchers.IO) {
                    client.newCall(request).execute()
                }

                response.use { res ->
                    if (!res.isSuccessful) {
                        withContext(Dispatchers.Main) {
                            statusText.text = "APIエラー: ${res.code}"
                            resetReplies()
                            showLoading(false)
                        }
                        return@use
                    }

                    val responseBody = res.body?.string().orEmpty()
                    val obj = JSONObject(responseBody)
                    val summary =
                        if (obj.isNull("summary")) null else obj.optString("summary", null)
                    val replies = obj.optJSONArray("replies") ?: JSONArray()

                    withContext(Dispatchers.Main) {
                        statusText.text = "返信案ができました。タップでコピーしてください。"

                        if (tone == "night") {
                            summaryText.text = ""
                            summaryText.visibility = View.GONE
                        } else {
                            summaryText.visibility = View.VISIBLE
                            summaryText.text = summary ?: ""
                        }

                        replyButtons.forEachIndexed { index, button ->
                            val textReply = replies.optString(index, "")
                            if (textReply.isNotBlank()) {
                                button.text = textReply
                            } else {
                                button.text = "返信案${index + 1}（なし）"
                            }
                        }
                        // ここで返信案コンテナを初めて表示
                        replyContainer.visibility = View.VISIBLE

                        showLoading(false)
                    }
                }
            } catch (e: Exception) {
                withContext(Dispatchers.Main) {
                    statusText.text = "通信エラーが発生しました: ${e.localizedMessage}"
                    resetReplies()
                    showLoading(false)
                }
            }
        }
    }

    private fun copyToClipboard(text: String) {
        val clipboard = getSystemService(Context.CLIPBOARD_SERVICE) as ClipboardManager
        clipboard.setPrimaryClip(ClipData.newPlainText("reply", text))
    }

    override fun onPause() {
        super.onPause()
        // トーン選択とスタイルメモを保存（念のため onPause でも保存）
        val prefs = getSharedPreferences(PREF_NAME, Context.MODE_PRIVATE)
        prefs.edit()

            .putString(KEY_STYLE_MEMO, userStyleEditText.text.toString())
            .apply()
    }

    override fun onDestroy() {
        super.onDestroy()
        cancel() // CoroutineScope を解放
    }
}
