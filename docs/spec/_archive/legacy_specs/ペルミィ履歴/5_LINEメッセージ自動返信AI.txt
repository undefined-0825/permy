1. 「自分のLINEメッセージをPythonで取得」はどこまで可能か
A. 公式に用意されている方法
LINE Messaging API（ボット用）

自分で「LINE公式アカウント」を作成して、Messaging API を有効化。

そのボット宛に送られたメッセージは、

LINE → あなたのWebhook URL（自前サーバ）にHTTP POST

WebhookのJSONをPythonで受け取る

これなら「ボットに届くメッセージ」はリアルタイムにPythonで取得できます。

トーク履歴のエクスポート（手動）

スマホのLINEアプリで、特定のトークを「テキストでエクスポート」できます。

それをPCに持ってきて、Pythonでテキストパースすれば、過去ログを解析したりAIに食わせたりは可能。

B. 公式に用意されていない／グレー・NGな方法
メモリスキャン・逆解析・非公開API・画面スクレイピングで
「個人LINEアプリの中身を自動収集する」のは、

セキュリティリスクが高い

LINEの利用規約違反になる可能性が高い

これはおすすめできないので、ここはやらない前提で話します。

2. ちょびさんの目的：「自動返信するAI」を作る構成
やりたいことは、

LINEのメッセージ → Python → AIで返信文生成 → LINEに返信

だと思うので、王道構成はこうなります：

LINE公式アカウントを作る

Messaging APIを有効化

Webhook URLを自分のサーバ（またはクラウドFunctions）に設定

PythonでWebhook受信用のWebアプリを作る（Flask / FastAPIなど）

受け取ったメッセージをAI（例: OpenAI API）に投げて返信文を生成

PythonからLINEの「reply API」を叩いてユーザーに返す

全体のイメージ
ユーザー → LINE公式アカウント(ボット)
       → [Webhook] あなたのPythonサーバ
       → AIで返信文生成
       → [Reply API] LINE公式アカウント
       → ユーザーに自動返信

3.1 Webhook受信（FastAPI例）
from fastapi import FastAPI, Request
import httpx
import os

LINE_CHANNEL_ACCESS_TOKEN = os.environ["LINE_CHANNEL_ACCESS_TOKEN"]

app = FastAPI()

async def reply_message(reply_token: str, text: str):
    url = "https://api.line.me/v2/bot/message/reply"
    headers = {
        "Authorization": f"Bearer {LINE_CHANNEL_ACCESS_TOKEN}",
        "Content-Type": "application/json"
    }
    body = {
        "replyToken": reply_token,
        "messages": [{"type": "text", "text": text}]
    }
    async with httpx.AsyncClient() as client:
        await client.post(url, json=body, headers=headers)

@app.post("/callback")
async def callback(request: Request):
    body = await request.json()
    events = body.get("events", [])

    for event in events:
        etype = event.get("type")
        if etype != "message":
            continue

        message = event.get("message", {})
        if message.get("type") != "text":
            continue

        user_text = message.get("text", "")
        reply_token = event.get("replyToken")

        # ここで AI に投げて返信文生成
        ai_reply = await generate_reply(user_text)

        await reply_message(reply_token, ai_reply)

    return "ok"

import openai

async def generate_reply(user_text: str) -> str:
    # ここは好きなモデル・APIでOK（例: OpenAI ChatCompletions）
    response = openai.ChatCompletion.create(
        model="gpt-4.1-mini",
        messages=[
            {"role": "system", "content": "あなたは丁寧なチャット用アシスタントです。"},
            {"role": "user", "content": user_text}
        ]
    )
    return response["choices"][0]["message"]["content"]

import sqlite3
import re
from datetime import datetime
from pathlib import Path
from typing import List, Tuple, Optional

# ==== 設定ここから ====

DB_PATH = "line_chat.db"

# LINEエクスポートでの自分側の名前と相手側の名前
# 実際のトーク履歴に合わせて変更してください。
MY_NAME = "ちょび"
PARTNER_NAME = "相手"

# 直近何件のメッセージをコンテキストに含めるか
CONTEXT_MESSAGE_COUNT = 30

# ==== 設定ここまで ====


# ==============================
# DB 初期化
# ==============================
def init_db(db_path: str = DB_PATH) -> None:
    conn = sqlite3.connect(db_path)
    cur = conn.cursor()

    # 今回は相手1人前提なので conversations は1レコードだけでもよいが、
    # 将来複数相手を扱いたくなることを見越してテーブルを切っておく。
    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS conversations (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL UNIQUE
        )
        """
    )

    cur.execute(
        """
        CREATE TABLE IF NOT EXISTS messages (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            conversation_id INTEGER NOT NULL,
            ts TEXT NOT NULL,
            speaker TEXT NOT NULL, -- 'me' or 'partner'
            raw_name TEXT NOT NULL, -- LINE上の表示名
            text TEXT NOT NULL,
            source TEXT NOT NULL, -- 'export' or 'webhook' など
            FOREIGN KEY (conversation_id) REFERENCES conversations(id)
        )
        """
    )

    conn.commit()
    conn.close()


def get_or_create_conversation_id(name: str, db_path: str = DB_PATH) -> int:
    conn = sqlite3.connect(db_path)
    cur = conn.cursor()

    cur.execute("SELECT id FROM conversations WHERE name = ?", (name,))
    row = cur.fetchone()
    if row:
        conv_id = row[0]
    else:
        cur.execute("INSERT INTO conversations (name) VALUES (?)", (name,))
        conv_id = cur.lastrowid
        conn.commit()

    conn.close()
    return conv_id


# ==============================
# LINEエクスポートテキストのパース
# ==============================

LINE_EXPORT_RE = re.compile(
    r"""
    ^(\d{4}/\d{1,2}/\d{1,2})  # 日付 2025/01/01
    \(.+?\)\s+                # (水) 等の曜日
    (\d{1,2}:\d{2})\s-\s      # 時刻 12:34 -
    (?:
        ([^:]+):\s            # 話者名 ちょび:
    )?
    (.*)$                     # メッセージ本体
    """,
    re.VERBOSE,
)


def parse_line_export_line(line: str) -> Optional[Tuple[datetime, str, str]]:
    """
    1行のテキストをパースして (日時, 発言者名, メッセージ) を返す。
    パースできない行（システムメッセージなど）は None を返す。
    """
    line = line.rstrip("\n")
    m = LINE_EXPORT_RE.match(line)
    if not m:
        return None

    date_str, time_str, name, message = m.groups()

    # 話者名が無い場合（システムメッセージ等）は無視する
    if not name:
        return None

    dt = datetime.strptime(f"{date_str} {time_str}", "%Y/%m/%d %H:%M")
    return dt, name, message


def import_line_export(
    txt_path: str,
    conversation_name: str = PARTNER_NAME,
    db_path: str = DB_PATH,
) -> None:
    """
    LINEエクスポートテキストを読み込み、messagesテーブルに保存する。
    """
    path = Path(txt_path)
    if not path.exists():
        raise FileNotFoundError(txt_path)

    conv_id = get_or_create_conversation_id(conversation_name, db_path)

    conn = sqlite3.connect(db_path)
    cur = conn.cursor()

    with path.open("r", encoding="utf-8") as f:
        for line in f:
            parsed = parse_line_export_line(line)
            if not parsed:
                continue

            dt, raw_name, message = parsed

            if raw_name == MY_NAME:
                speaker = "me"
            elif raw_name == PARTNER_NAME:
                speaker = "partner"
            else:
                # 念のためその他の名前はスキップ
                continue

            cur.execute(
                """
                INSERT INTO messages (conversation_id, ts, speaker, raw_name, text, source)
                VALUES (?, ?, ?, ?, ?, ?)
                """,
                (conv_id, dt.isoformat(), speaker, raw_name, message, "export"),
            )

    conn.commit()
    conn.close()


# ==============================
# コンテキスト取得＆プロンプト生成
# ==============================

def fetch_recent_messages(
    conversation_name: str = PARTNER_NAME,
    limit: int = CONTEXT_MESSAGE_COUNT,
    db_path: str = DB_PATH,
) -> List[Tuple[str, str, str]]:
    """
    直近 limit 件のメッセージを時系列順に返す。
    戻り値: List[(ts_iso, speaker, text)]
    """
    conn = sqlite3.connect(db_path)
    cur = conn.cursor()

    cur.execute("SELECT id FROM conversations WHERE name = ?", (conversation_name,))
    row = cur.fetchone()
    if not row:
        conn.close()
        return []
    conv_id = row[0]

    cur.execute(
        """
        SELECT ts, speaker, text
        FROM messages
        WHERE conversation_id = ?
        ORDER BY ts DESC
        LIMIT ?
        """,
        (conv_id, limit),
    )
    rows = cur.fetchall()
    conn.close()

    # 直近を取ってきているので、昇順に並べ直す
    rows.reverse()
    return rows

def main():
    import argparse

    parser = argparse.ArgumentParser(description="LINE過去ログから返信案を生成するPoC")
    subparsers = parser.add_subparsers(dest="command")

    # 1) DB初期化
    sp_init = subparsers.add_parser("init-db", help="SQLite DB を初期化する")

    # 2) エクスポート取り込み
    sp_import = subparsers.add_parser("import", help="LINEエクスポートTXTを取り込む")
    sp_import.add_argument("txt_path", help="LINEエクスポートTXTのパス")

    # 3) 返信案生成
    sp_reply = subparsers.add_parser("reply", help="標準入力のテキストに対する返信案を生成")
    # incoming_text は標準入力から読むので引数は不要

    args = parser.parse_args()

    if args.command == "init-db":
        init_db()
        print("DB初期化完了:", DB_PATH)
    elif args.command == "import":
        init_db()
        import_line_export(args.txt_path)
        print("取り込み完了:", args.txt_path)
    elif args.command == "reply":
        print("相手からの新規メッセージを入力してください（Ctrl+Dで確定）:")
        incoming_text = "".join(iter(input, ""))  # 単純な読み方でOKならこれでもよい
        # 上記だとWindowsで扱いにくいので、単行でいいなら input() 一発でもOK
        if not incoming_text.strip():
            # 単行入力用にフォールバック
            incoming_text = input().strip()

        result = suggest_reply(incoming_text)
        print("=== 返信案 ===")
        print(result)
    else:
        parser.print_help()

3.1 persona（ペルソナ定義）
CREATE TABLE persona (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    name            TEXT NOT NULL,              -- 仕事用 / 営業用 / 家庭用 など
    description     TEXT,                       -- 自分用メモ
    is_default      INTEGER NOT NULL DEFAULT 0, -- 0/1
    sort_order      INTEGER NOT NULL DEFAULT 0, -- 表示順

    created_at      TEXT NOT NULL,
    updated_at      TEXT NOT NULL
);
「人格名」+「自分用メモ」を持つシンプルな定義

文体系は別テーブル persona_style に切り出しておくと拡張が楽

3.2 persona_style（文体・トーンなど）
CREATE TABLE persona_style (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    persona_id      INTEGER NOT NULL UNIQUE, -- 1:1 を想定
    tone            TEXT NOT NULL,           -- 'formal', 'friendly', 'playful' など
    length_pref     TEXT NOT NULL,           -- 'short', 'normal', 'long'
    emoji_level     INTEGER NOT NULL,        -- 0:使わない, 1:控えめ, 2:普通, 3:多め
    politeness      INTEGER NOT NULL,        -- 1〜5 想定：敬語の強さ
    directness      INTEGER NOT NULL,        -- 1〜5：ストレートさ
    smalltalk_level INTEGER NOT NULL,        -- 1〜5：雑談をどれくらい入れるか

    -- 将来拡張用のフリーフィールド
    extra_json      TEXT,                    -- JSONで細かいパラメータを持つ

    created_at      TEXT NOT NULL,
    updated_at      TEXT NOT NULL,

    FOREIGN KEY (persona_id) REFERENCES persona(id)
);
LLM プロンプトにそのまま落とせる粒度のパラメータ

extra_json は将来「業種タグ」「口癖」などを追加するための逃げ道

3.3 contact（相手）
CREATE TABLE contact (
    id                  INTEGER PRIMARY KEY AUTOINCREMENT,
    display_name        TEXT NOT NULL,     -- 表示名（LINE名など）
    alias_name          TEXT,              -- 自分用の呼び方（打ち合わせA社担当…など）
    tags                TEXT,              -- カンマ区切り or JSON（'上司,重要顧客' 等）

    default_persona_id  INTEGER,           -- この相手のデフォルトペルソナ

3.5 ng_phrase（NGワード・禁止表現）
CREATE TABLE ng_phrase (

【役割】
- あなたは「自動返信ツール」ではなく、「人間が最終確認して送信するための、返信案のアイデア出し担当」です。
- ユーザーが設定したペルソナ・相手との関係性・NG表現を尊重して、返信案を複数パターン提示してください。

【守るべき原則】

7. ペルソナの設定（丁寧さ・長さ・絵文字レベルなど）に合わない文体は避ける。

【出力形式】
- 常に3パターンの返信案を出してください。
  - A: 短めで要点だけまとめた案
  - B: 標準的な長さの案

5. StateNotifierの基本形
5.1 状態クラス（immutable）
class PersonaState {
  final bool loading;
  final List<Persona> personas;
  final int? selectedPersonaId;

  const PersonaState({
    this.loading = false,
    this.personas = const [],
    this.selectedPersonaId,
  });

  PersonaState copyWith({
    bool? loading,
    List<Persona>? personas,
    int? selectedPersonaId,
  }) {
    return PersonaState(
      loading: loading ?? this.loading,
      personas: personas ?? this.personas,
      selectedPersonaId: selectedPersonaId ?? this.selectedPersonaId,
    );
  }
}
👉

状態は 必ず immutable

変更は copyWith で行う

5.2 Notifier（ロジック）
class PersonaNotifier extends StateNotifier<PersonaState> {
  PersonaNotifier() : super(const PersonaState());

  Future<void> loadPersonas() async {
    state = state.copyWith(loading: true);

    final list = await fetchFromDb();

    state = state.copyWith(
      loading: false,
      personas: list,
    );
  }

  void selectPersona(int id) {
    state = state.copyWith(selectedPersonaId: id);
  }
}
5.3 Provider定義
final personaProvider =
    StateNotifierProvider<PersonaNotifier, PersonaState>(
  (ref) => PersonaNotifier(),
);

このスレッドの要約（引き継ぎ用）
1. プロダクト概要
目的
複数の「ペルソナ（場面ごとの自分）」を明示的に切り替え、
LINE等のメッセージ返信をストレスなく考えるための思考補助アプリ。

特徴

自動送信なし（提案のみ）

AIが主語ではなく「ユーザーが主語」

返信の最適解を断定せず、複数案を提示

マルチペルソナ切替が中核価値

ターゲット

夜職（営業LINEを日常的に送る）

ビジネスマン（顧客・上司・関係者対応）

複数の役割を持つ個人ユーザー

課金

月額サブスク 980円

1か月無料トライアル

2. UXの基本方針（重要）
LINE等のチャットアプリには介入しない

OS標準の共有シート（Share）からテキストを取り込む

ユーザー操作を常に起点にする（特許・規約・審査対策）

UXフロー
初回起動（コンセプト説明・自動送信しないことを明示）

ペルソナ登録（最低1件）

相手プロファイル登録（任意）

LINE等から共有シートでメッセージ取り込み

ペルソナ確認・切替

返信提案生成（3案）

コピーして元アプリへ戻る

3. 技術スタック
フロント：Flutter

言語：Dart

状態管理：Riverpod

DB：SQLite（drift想定）

AI：外部API（推論のみ）

自動送信：なし

Flutter + Dart を選択した理由：

宣言的UI（UI = f(State)）

ペルソナ切替・状態駆動と相性が良い

iOS / Android 両対応

共有シート実装が容易

4. 状態管理の設計思想
デザインパターン的には
Observer + State + Command のハイブリッド

UI（Widget）は状態を監視するだけ

状態変更は Notifier 経由のみ

主要な状態単位
PersonaState（ペルソナ一覧・選択中）

ContactState（相手情報）

SharedTextState（共有されたテキスト）

ReplyState（返信生成中／結果）

SubscriptionState（課金状態）
