    # ここに既存の OpenAI 呼び出しロジック
    # tone による system_prompt 切替もそのまま
    ...

if __name__ == "__main__":
    import uvicorn

    port = int(os.getenv("PORT", "8000"))
    uvicorn.run("main:app", host="0.0.0.0", port=port)
Render では **「Start command」で uvicorn を叩く」**ので、

1. 無難な解決策：pull → push（履歴を統合）

git branch         # 今いるブランチ名確認（main or master）

git pull --rebase origin main
# 必要あればコンフリクト解消 → add → continue
git push origin main
ポイント
--rebase を付けると履歴がきれいに直列になる

コンフリクトが出ても、ほぼ README と被るだけなので簡単に直せることが多い

import os
import logging
from typing import List

from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, Field
from openai import OpenAI, OpenAIError

# ---------------------------
# ログ設定
# ---------------------------
logger = logging.getLogger("talk_assist")
logging.basicConfig(level=logging.INFO)

# ---------------------------
# OpenAI クライアント
# ---------------------------
api_key = os.getenv("OPENAI_API_KEY")
if not api_key:
    # 起動時に分かりやすく落とす
    raise RuntimeError("OPENAI_API_KEY is not set in environment variables.")

client = OpenAI(api_key=api_key)

MODEL_NAME = "gpt-4.1-mini"

# ---------------------------
# FastAPI 本体
# ---------------------------
app = FastAPI(
    title="Talk Assist API",
    description="LINEトーク要約＆返信生成API",
    version="0.1.0",
)

# CORS（将来のWeb対応も考慮して緩めに）
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # 本番では必要なドメインに絞る
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],

# モデル定義

class TalkRequest(BaseModel):
    text: str = Field(..., description="LINE等のトーク履歴")
    tone: str = Field(..., description="standard | night | business")


class TalkResponse(BaseModel):
    summary: str
    replies: List[str]


# ---------------------------
# プロンプト設計
# ---------------------------

def build_system_prompt(tone: str) -> str:
    """
    tone ごとに system プロンプトを切り替える。
    出力フォーマットは共通で、必ず以下の形式にさせる：

    1行目: 会話の要約（1〜3行程度）
    2行目以降: 「- 」で始まる返信候補を3件
    """
    base = """
あなたは日本語でLINE等のメッセージに対する「要約＋返信候補」を作るアシスタントです。

前提ルール:
- 自動送信はしない。あくまで「ユーザーが送るための案」を出す。
- 相手との関係性（友人・同僚・顧客・指名客など）や温度感を会話から推定する。
- 日本語として自然で、読みやすく、誤解を生まない文にする。
- 相手を不快にさせる表現や、過度に強い要求は避ける。

出力フォーマット（厳守）:
1行目: 会話全体の要約（1〜3文）
2行目以降: 返信候補を3つ、各行を「- 」で始めて出力する。
例:
要約文………
- 返信案1
- 返信案2
- 返信案3

トーク履歴の文面はそのまま繰り返さず、要点を抽出して要約すること。
"""

    if tone == "business":
        extra = """
トーン: business

- 上司・顧客・取引先などに送ることを想定する。
- 丁寧語・敬語を基本としつつ、回りくどくなりすぎないようにする。
- クッション言葉（「もしよろしければ」「お手数ですが」など）を適度に使う。
- 感情的・攻撃的にならない。
"""
    elif tone == "night":
        extra = """
トーン: night

- 夜職（キャバクラ・ラウンジ・ホストなど）の営業LINEを想定する。
- フレンドリーで親しみやすいが、軽すぎて失礼にならないバランスにする。
- 相手の負担にならない範囲で「来店・指名・予約」につながる自然な流れを意識する。
- 押し売り感を出さず、「相手の都合や気持ちを尊重する」言い回しを必ず入れる。
- 既存客には「前回のお礼」や「会いに来てくれたら嬉しい」など、関係維持・向上の要素を盛り込む。
"""
    else:
        # default: standard
        extra = """
トーン: standard

- 友人・知人との日常的なやりとりを想定する。
- 敬語とタメ口の中間くらいの、自然で話しやすい日本語にする。
- きつい言い回しや、相手を責める表現は避ける。
"""

    return base + extra


def build_user_prompt(text: str) -> str:
    """
    user に渡すプロンプト。
    ここでは「要約＋返信3案」を明示。
    """
    return f"""
以下はLINE等のトーク履歴です。この会話を読み、要約と返信候補3つを作成してください。

【出力要件】
- 1行目に会話全体の要約（1〜3文）
- 2行目以降に「- 」で始まる返信案を3つ
- 返信案は、それぞれニュアンスが少しずつ異なるようにする
- 返信案は、そのまま送信できる文にする

【トーク履歴】
{text}
"""


def parse_model_output(raw: str) -> TalkResponse:
    """
    モデルから返ってきたテキストを
    summary / replies[3] に分解する。
    """
    lines = [line.strip() for line in raw.splitlines() if line.strip()]

    if not lines:
        raise ValueError("Empty response from model")

    summary = lines[0]

    replies: List[str] = []
    for line in lines[1:]:
        if line.startswith("- "):
            replies.append(line[2:].strip())
        elif line.startswith("-"):
            replies.append(line[1:].strip())

    # 3件未満の場合は空文字で埋めて長さ3に合わせる
    while len(replies) < 3:
        replies.append("")

    # 3件より多ければ3件に絞る
    replies = replies[:3]

    return TalkResponse(summary=summary, replies=replies)

    return {"status": "ok"}


@app.post("/api/talk/assist", response_model=TalkResponse)
async def talk_assist(req: TalkRequest) -> TalkResponse:
    if req.tone not in ("standard", "night", "business"):
        raise HTTPException(status_code=400, detail="invalid tone")

    if not req.text or not req.text.strip():
        raise HTTPException(status_code=400, detail="text is empty")

    logger.info(
        "request: len=%d tone=%s",
        len(req.text),
        req.tone,
    )

    system_prompt = build_system_prompt(req.tone)
    user_prompt = build_user_prompt(req.text)

    try:
        response = client.responses.create(
            model=MODEL_NAME,
            input=[
                {
                    "role": "system",
                    "content": [
                        {"type": "text", "text": system_prompt},
                    ],
                },
                {
                    "role": "user",
                    "content": [
                        {"type": "text", "text": user_prompt},
                    ],
                },
            ],
        )
    except OpenAIError as e:
        logger.exception("OpenAI API error")
        raise HTTPException(status_code=502, detail=f"OpenAI API error: {e}")

    try:
        # 新Responses APIの想定構造: output[0].content[0].text
        raw_text = response.output[0].content[0].text
    except Exception as e:
        logger.exception("Unexpected response format from OpenAI")
        raise HTTPException(status_code=500, detail=f"Unexpected response format: {e}")

def preprocess_text(raw_text: str, tone: ToneLiteral) -> Tuple[str, str, int, int]:
    """
    長文対策の前処理。
    戻り値: (used_text, mode, original_len, used_len)
      mode: "full" | "trimmed" | "summarized"
    """
    text = raw_text.strip()
    original_len = len(text)

    # 生テキストが短ければそのまま
    if original_len <= MAX_USED_CHARS:
        return text, "full", original_len, original_len

    # まずは頭 + 尻でトリミング
    head = text[: MAX_USED_CHARS // 2]
    tail = text[-MAX_USED_CHARS // 2 :]
    trimmed = head + "\n...\n" + tail

    if original_len <= MAX_RAW_CHARS:
        used_len = len(trimmed)
        return trimmed, "trimmed", original_len, used_len

    # さらに長い場合は要約フェーズ（summarized）
    try:
        summary = summarize_conversation(text, tone)
        used_text = summary
        mode = "summarized"
    except Exception as e:
        # 要約に失敗したらトリム版で妥協
        print(f"[WARN] summarize_conversation failed: {e}")
        used_text = trimmed
        mode = "trimmed-fallback"

    used_len = len(used_text)
    return used_text, mode, original_len, used_len


# ========== 
# メインエンドポイント 
# ==========

def parse_ai_output(content: str) -> Tuple[str, List[str]]:
    """
    モデルから返ってきたテキストを「要約」「返信案」に分解する。
    - '要約：' / '要約:' 行は見出しとして扱う
    - '返信案：' / '返信案:' 行は見出しとしてスキップ
    - '- ' や '・' で始まる行を返信候補として優先
    """
    lines = [line.strip() for line in content.splitlines() if line.strip()]
    if not lines:
        return "（AIからの応答が空でした）", []

    summary: str | None = None
    summary_index: int = -1

    # 要約行を探す
    for idx, line in enumerate(lines):
        # "要約：" だけの行（中身なし）は見出し扱いでスキップ
        if re.fullmatch(r"要約[:：]?", line):
            continue

        # "要約: XXX" 形式なら、コロン以降を要約として取る
        m = re.match(r"要約[:：]\s*(.+)", line)
        if m:
            summary = m.group(1).strip()
            summary_index = idx
            break

        # それ以外は最初に出てきた非空行を要約とみなす
        summary = line
        summary_index = idx
        break

    if summary is None:
        summary = "（要約を抽出できませんでした）"
        summary_index = -1

    replies: List[str] = []

    # 返信候補を探す
    start_idx = summary_index + 1
    for line in lines[start_idx:]:
        # "返信案" 見出しっぽい行はスキップ
        if re.fullmatch(r"返信案[:：]?", line):
            continue

        # "- " や "・" で始まる行を優先
        if line.startswith("-"):
            replies.append(line.lstrip("-").strip())
            continue
        if line.startswith("・"):
            replies.append(line.lstrip("・").strip())
            continue

        # "1. xxx" / "１）xxx" など番号付きも拾う
        if re.match(r"^[0-9０-９]+[.)．）]\s*", line):
            line2 = re.sub(r"^[0-9０-９]+[.)．）]\s*", "", line)
            replies.append(line2.strip())
            continue

        # それ以外: 既に返信候補があれば無視、まだ無ければ拾っておく
        if not replies:
            replies.append(line)

    # 最大3件
    replies = replies[:3]

    return summary, replies

@app.post("/api/talk/assist", response_model=TalkResponse)
async def talk_assist(req: TalkRequest) -> TalkResponse:
    raw_text = (req.text or "").strip()
    tone_raw = (req.tone or "standard").lower()

    if not raw_text:
        return TalkResponse(summary="（入力が空です）", replies=[])

    # tone を正規化
    if tone_raw not in ("standard", "night", "business"):
        tone: ToneLiteral = "standard"
    else:
        tone = tone_raw  # type: ignore[assignment]

    # 長文前処理
    used_text, mode, original_len, used_len = preprocess_text(raw_text, tone)

    system_prompt = build_system_prompt(tone)
    user_prompt = build_user_prompt(used_text, mode)
    _, temperature = build_tone_desc_and_temp(tone)

    try:
        completion = client.responses.create(
            model=MODEL_NAME,
            input=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_prompt},
            ],
            temperature=temperature,
            max_output_tokens=512,
        )

        # Responses API の結果からテキストを取り出す
        content = completion.output[0].content[0].text.strip()

        summary, replies = parse_ai_output(content)

        if not replies:
            # 返信案がうまく取れなかったケースはログに残しておく
            write_log(
                {
                    "event": "parsed_without_replies",
                    "tone": tone,
                    "original_len": original_len,
                    "used_len": used_len,
                    "mode": mode,
                    "raw_content_preview": content[:120],
                }
            )
        write_log(
            {
                "event": "success",
                "tone": tone,
                "original_len": original_len,
                "used_len": used_len,
                "mode": mode,
                "summary_preview": summary[:50],
                "replies_count": len(replies),
            }
        )

        return TalkResponse(summary=summary, replies=replies)

    except Exception as e:
        # サーバ側ログ
        print(f"[ERROR] talk_assist OpenAI error: {e}")

                "event": "error",

        return TalkResponse(

    const textarea = document.getElementById('inputText');
    const charCount = document.getElementById('charCount');
    const generateBtn = document.getElementById('generateBtn');
    const statusEl = document.getElementById('status');
    const resultEl = document.getElementById('result');
    const summaryText = document.getElementById('summaryText');
    const replyButtons = document.querySelectorAll('.reply-btn');

    // 文字数カウント
    textarea.addEventListener('input', () => {
      const len = textarea.value.length;
      charCount.textContent = len.toString();
    });

    function getTone() {
      const radios = document.querySelectorAll('input[name="tone"]');
      for (const r of radios) {
        if (r.checked) return r.value;
      }
      return 'standard';
    }

    async function callApi() {
      const text = textarea.value.trim();
      if (!text) {
        statusEl.textContent = 'トーク履歴を入力してください。';
        return;
      }

      generateBtn.disabled = true;
      statusEl.textContent = '生成中...';
      resultEl.style.display = 'none';

      // 同一オリジンの /api/talk/assist を叩く
      const url = '/api/talk/assist';

      try {
        const res = await fetch(url, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({
            text: text,
            tone: getTone(),
          }),
        });

        if (!res.ok) {
          const bodyText = await res.text();
          console.error('API error', res.status, bodyText);
          statusEl.textContent = `APIエラー: ${res.status}`;
          return;
        }

        const data = await res.json();
        summaryText.textContent = data.summary || '';

        const replies = data.replies || [];
        replyButtons.forEach((btn, idx) => {
          const text = replies[idx] || '';
          btn.textContent = text || `返信候補${idx + 1}（未生成）`;
          btn.disabled = !text;
          btn.classList.remove('copied');
        });

        resultEl.style.display = 'block';
        statusEl.textContent = '生成完了。返信ボタンをタップするとコピーできます。';
      } catch (err) {
        console.error(err);
        statusEl.textContent = '通信エラーが発生しました。時間をおいて再試行してください。';
      } finally {
        generateBtn.disabled = false;
      }
    }

    generateBtn.addEventListener('click', () => {
      callApi();
    });

    // 返信ボタン → クリップボードコピー
    replyButtons.forEach((btn) => {
      btn.addEventListener('click', async () => {
        const text = btn.textContent || '';
        if (!text || text.startsWith('返信候補')) return;

        try {
          await navigator.clipboard.writeText(text);
          replyButtons.forEach((b) => b.classList.remove('copied'));
          btn.classList.add('copied');
          statusEl.textContent = '返信案をクリップボードにコピーしました。';
        } catch (err) {
          console.error(err);
          statusEl.textContent = 'コピーに失敗しました。手動で選択してください。';
        }
      });
    });

# プロジェクトルート / static パス
BASE_DIR = Path(__file__).resolve().parent
STATIC_DIR = BASE_DIR / "static"

# /static で静的ファイル配信
if STATIC_DIR.exists():
    app.mount("/static", StaticFiles(directory=str(STATIC_DIR)), name="static")


@app.get("/", response_class=HTMLResponse)
async def index():
    """
    シンプルなWeb UIを返す。
    /backend/static/index.html が存在しなければメッセージだけ出す。
    """
    index_file = STATIC_DIR / "index.html"
    if index_file.exists():
        return index_file.read_text(encoding="utf-8")
    return HTMLResponse(
        "<h1>Talk Assist Backend</h1><p>static/index.html が配置されていません。</p>",
        status_code=200,
    )

      try {

          statusEl.textContent = 'コピーされたテキストが見つかりません。LINE側でコピーされているか確認してください。';

      const text = textarea.value.trim();

1. 何を前提にするか
✅ 全履歴は要らない
モデル的には「直近のやり取り 10〜30通」ぐらいで十分コンテキストになる

なので “全部選択できない＝詰み” ではない

✅ ユーザーがやる操作
パターンを 2つ 想定しておけばOK：

かんたんモード：直近の会話だけコピー

ガチモード：トーク履歴の書き出し（必要なら）

ただ、今テストしたいのは「日常でのサクッと利用」だから、
まずは 1. かんたんモード を前提で説明した方が早い。

2. 実際のLINEでの操作フロー（かんたんモード）
Android / iOS 共通イメージ（今のUI）
直近のやり取りだけでOK という前提にした説明：

LINEでトーク画面を開く

要約したいところの一番上の吹き出しを長押し

メニューから 「部分コピー」 をタップ

画面下に出る範囲選択で、下にドラッグして必要なところまで選ぶ

右下の 「コピー」 を押す

ブラウザに戻って、「貼り付け」ボタン or 長押し → 貼り付け

トーンを選んで「生成」

※「部分コピー」がわかりにくい人には、
最悪 “重要な数通だけ普通のコピーで何回か貼る” でも動くようにしておけばいい。

3. トーク履歴フル出力のルート（ガチモード）
これは “ヘビーユーザーや長文相談向け” 用のオプションとしてヘルプに書く感じ。

（機種やバージョンで多少違うけど、概ね）

トーク画面右上の 三本線（≡） をタップ

「その他」 or 「設定」→ 「トーク履歴を送信」

「テキストで送信」を選ぶ

「自分のメール」か「Keep」へ送る

            <intent-filter>
                <action android:name="android.intent.action.MAIN" />
                <category android:name="android.intent.category.LAUNCHER" />
            </intent-filter>

            <!-- LINEなどからの「共有 / トーク履歴を送信」を受け取る -->
            <intent-filter>
                <action android:name="android.intent.action.SEND" />
                <category android:name="android.intent.category.DEFAULT" />
                <data android:mimeType="text/plain" />
            </intent-filter>

            <!-- .txtファイルで送られてくる場合にも対応したければオプションで -->
            <intent-filter>
                <action android:name="android.intent.action.SEND" />

                <data android:mimeType="text/*" />
            </intent-filter>

        </activity>

    </application>

        inputText = findViewById(R.id.inputText)
        toneGroup = findViewById(R.id.toneGroup)
        statusText = findViewById(R.id.statusText)
        summaryText = findViewById(R.id.summaryText)
        replyButtons = listOf(
            findViewById(R.id.reply1Button),
            findViewById(R.id.reply2Button),
            findViewById(R.id.reply3Button),
        )

        // 共有から渡ってきたテキストを取得
        handleShareIntent()

        findViewById<Button>(R.id.clearButton).setOnClickListener {
            inputText.setText("")
            summaryText.text = ""
            replyButtons.forEach { it.text = "返信案" }
        }

        findViewById<Button>(R.id.generateButton).setOnClickListener {
            callApi()
        }

        // 返信ボタン → クリップボードコピー
        replyButtons.forEach { btn ->
            btn.setOnClickListener {
                val text = btn.text.toString()
                if (text.isNotBlank() && !text.startsWith("返信案")) {
                    copyToClipboard(text)
                    statusText.text = "返信案をコピーしました。LINEに貼り付けて送ってください。"
                }
            }
        }
    }

    private fun handleShareIntent() {
        val action = intent?.action
        val type = intent?.type
        if (action == android.content.Intent.ACTION_SEND && type != null) {
            if (type.startsWith("text/")) {
                val sharedText = intent.getStringExtra(android.content.Intent.EXTRA_TEXT)
                if (!sharedText.isNullOrBlank()) {
                    inputText.setText(sharedText)
                    statusText.text = "LINEから受け取ったトーク履歴を読み込みました。"
                }
            }
            // .txtファイルとして送られるケースへの対応を入れたければここでStreamを読む
        }
    }

    private fun getSelectedTone(): String {
        return when (toneGroup.checkedRadioButtonId) {
            R.id.toneNight -> "night"
            R.id.toneBusiness -> "business"
            else -> "standard"
        }
    }

    private fun callApi() {
        val text = inputText.text.toString().trim()
        if (text.isBlank()) {
            statusText.text = "トーク内容が空です。LINEから共有してから実行してください。"
            return
        }

        statusText.text = "生成中..."
        summaryText.text = ""
        replyButtons.forEach { it.text = "生成中..." }

        val json = JSONObject().apply {
            put("text", text)
            put("tone", getSelectedTone())
        }

        val body = RequestBody.create(
            "application/json; charset=utf-8".toMediaTypeOrNull(),
            json.toString()
        )

        val request = Request.Builder()
            .url("$apiBaseUrl/api/talk/assist")
            .post(body)
            .build()

        launch {
            withContext(Dispatchers.IO) {
                client.newCall(request).execute()
            }.use { response ->
                if (!response.isSuccessful) {
                    withContext(Dispatchers.Main) {
                        statusText.text = "APIエラー: ${response.code}"
                    }
                    return@use
                }

                val responseBody = response.body?.string().orEmpty()
                val obj = JSONObject(responseBody)
                val summary = obj.optString("summary", "")
                val replies = obj.optJSONArray("replies") ?: JSONArray()

                withContext(Dispatchers.Main) {
                    summaryText.text = summary
                    for (i in replyButtons.indices) {
                        val textReply = replies.optString(i, "")
                        if (textReply.isNotBlank()) {
                            replyButtons[i].text = textReply
                        } else {
                            replyButtons[i].text = "返信案${i + 1}（なし）"
                        }
                    }
                    statusText.text = "生成完了。返信案をタップしてコピーしてください。"
                }
            }
        }
    }

    private fun copyToClipboard(text: String) {
        val clipboard = getSystemService(Context.CLIPBOARD_SERVICE) as ClipboardManager
        clipboard.setPrimaryClip(ClipData.newPlainText("reply", text))
    }

    override fun onDestroy() {
        super.onDestroy()
        cancel() // CoroutineScope
    }

private fun handleShareIntent() {
    val action = intent?.action
    val type = intent?.type

    if (Intent.ACTION_SEND == action && type != null) {

        // 1) まずは普通のテキスト共有（EXTRA_TEXT）を試す
        if (type.startsWith("text/")) {
            val sharedText = intent.getStringExtra(Intent.EXTRA_TEXT)
            if (!sharedText.isNullOrBlank()) {
                inputText.setText(sharedText)
                statusText.text = "LINEから受け取ったトーク履歴を読み込みました。"
                return
            }
        }

        // 2) EXTRA_TEXT が無い場合は、ファイル共有（EXTRA_STREAM）を試す
        val uri: Uri? = intent.getParcelableExtra(Intent.EXTRA_STREAM)
        if (uri != null) {
            try {
                contentResolver.openInputStream(uri)?.use { input ->
                    val text = input.bufferedReader(Charsets.UTF_8).readText()
                    inputText.setText(text)
                    statusText.text = "LINEのトーク履歴ファイルを読み込みました。"
                }
            } catch (e: Exception) {
                e.printStackTrace()
                statusText.text = "トーク履歴ファイルの読み込みに失敗しました。"
            }
        }
    }
}

package com.example.talkassistshare

import android.content.ClipData
import android.content.ClipboardManager
import android.content.Context
import android.content.Intent
import android.net.Uri
import android.os.Bundle
import android.widget.Button
import android.widget.EditText
import android.widget.RadioGroup
import android.widget.TextView
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import kotlinx.coroutines.*
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.OkHttpClient
import okhttp3.Request
import okhttp3.RequestBody.Companion.toRequestBody
import org.json.JSONArray
import org.json.JSONObject

class MainActivity : AppCompatActivity(), CoroutineScope by MainScope() {

    private lateinit var inputText: EditText
    private lateinit var toneGroup: RadioGroup
    private lateinit var statusText: TextView
    private lateinit var summaryText: TextView
    private lateinit var replyButtons: List<Button>

    private val client = OkHttpClient()

    // Render のバックエンドURL
    private val apiBaseUrl = "https://talk-assist-backend.onrender.com"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        inputText = findViewById(R.id.inputText)
        toneGroup = findViewById(R.id.toneGroup)
        statusText = findViewById(R.id.statusText)
        summaryText = findViewById(R.id.summaryText)
        replyButtons = listOf(
            findViewById(R.id.reply1Button),
            findViewById(R.id.reply2Button),
            findViewById(R.id.reply3Button),
        )

        // LINEなどからの共有を処理
        handleShareIntent()

        findViewById<Button>(R.id.clearButton).setOnClickListener {
            inputText.setText("")
            summaryText.text = ""
            resetReplies()
            statusText.text = ""
        }

        findViewById<Button>(R.id.generateButton).setOnClickListener {
            callApi()
        }

        // 返信案ボタン → クリップボードコピー
        replyButtons.forEach { btn ->
            btn.setOnClickListener {
                val text = btn.text.toString()
                if (text.isNotBlank() && !text.startsWith("返信案")) {
                    copyToClipboard(text)
                    statusText.text = "返信案をコピーしました。LINEに貼り付けて送ってください。"
                    Toast.makeText(this, "コピーしました", Toast.LENGTH_SHORT).show()
                }
            }
        }
    }

    /**
     * LINEなどからの共有インテントを処理
     * - EXTRA_TEXT にテキストが入っている場合
     * - EXTRA_STREAM に .txt ファイルが渡される場合
     */
    private fun handleShareIntent() {
        val action = intent?.action
        val type = intent?.type

        if (Intent.ACTION_SEND == action && type != null) {

            // 1) 普通のテキスト共有（EXTRA_TEXT）
            if (type.startsWith("text/")) {
                val sharedText = intent.getStringExtra(Intent.EXTRA_TEXT)
                if (!sharedText.isNullOrBlank()) {
                    inputText.setText(sharedText)
                    statusText.text = "LINEから受け取ったトーク履歴を読み込みました。"
                    return
                }
            }

            // 2) ファイル共有（EXTRA_STREAM, 例: .txt）
            val uri: Uri? = intent.getParcelableExtra(Intent.EXTRA_STREAM)
            if (uri != null) {
                try {
                    contentResolver.openInputStream(uri)?.use { input ->
                        val text = input.bufferedReader(Charsets.UTF_8).readText()
                        inputText.setText(text)
                        statusText.text = "LINEのトーク履歴ファイルを読み込みました。"
                    }
                } catch (e: Exception) {
                    e.printStackTrace()
                    statusText.text = "トーク履歴ファイルの読み込みに失敗しました。"
                }
            }
        }
    }

    private fun getSelectedTone(): String {
        return when (toneGroup.checkedRadioButtonId) {
            R.id.toneNight -> "night"
            R.id.toneBusiness -> "business"
            else -> "standard"
        }
    }

    private fun resetReplies() {
        replyButtons.forEachIndexed { index, button ->
            button.text = "返信案${index + 1}"
        }
    }

    private fun callApi() {
        val text = inputText.text.toString().trim()
        if (text.isBlank()) {
            statusText.text = "トーク内容が空です。LINEから共有してから実行してください。"
            resetReplies()
            summaryText.text = ""
            return
        }

        statusText.text = "生成中..."
        summaryText.text = ""
        replyButtons.forEach { it.text = "生成中..." }

        // リクエストJSON
        val json = JSONObject().apply {
            put("text", text)
            put("tone", getSelectedTone())
        }

        // OkHttp4 推奨の書き方：toMediaType & toRequestBody を使用
        val body = json.toString()
            .toRequestBody("application/json; charset=utf-8".toMediaType())

        val request = Request.Builder()
            .url("$apiBaseUrl/api/talk/assist")
            .post(body)
            .build()

        launch {
            try {
                val response = withContext(Dispatchers.IO) {
                    client.newCall(request).execute()
                }

                response.use { res ->
                    if (!res.isSuccessful) {
                        withContext(Dispatchers.Main) {
                            statusText.text = "APIエラー: ${res.code}"
                            resetReplies()
                        }
                        return@use
                    }

                    val responseBody = res.body?.string().orEmpty()
                    val obj = JSONObject(responseBody)
                    val summary = obj.optString("summary", "")
                    val replies = obj.optJSONArray("replies") ?: JSONArray()

                    withContext(Dispatchers.Main) {
                        summaryText.text = summary
                        replyButtons.forEachIndexed { index, button ->
                            val textReply = replies.optString(index, "")
                            if (textReply.isNotBlank()) {
                                button.text = textReply
                            } else {
                                button.text = "返信案${index + 1}（なし）"
                            }
                        }
                        statusText.text = "生成完了。返信案をタップしてコピーしてください。"
                    }
                }
            } catch (e: Exception) {
                withContext(Dispatchers.Main) {
                    statusText.text = "通信エラーが発生しました: ${e.localizedMessage}"
                    resetReplies()
                }
            }
        }
    }

    private fun copyToClipboard(text: String) {
        val clipboard = getSystemService(Context.CLIPBOARD_SERVICE) as ClipboardManager
        clipboard.setPrimaryClip(ClipData.newPlainText("reply", text))
    }

    override fun onDestroy() {
        super.onDestroy()
        cancel() // CoroutineScope を解放
    }
}

1. 何が起こっているか（Render 側）
Render の Free プランは

一定時間アクセスが無いと コンテナを停止

次のアクセスで コンテナ起動（コールドスタート） が走る

この起動に 数十秒かかることがある

一方、こちらの Android アプリで使っている OkHttp のデフォルトは

connect / read timeout ≒ 10 秒

なので、

アプリからリクエスト
→ Render が「よし、コンテナ起動するわ…」とモタモタ
→ 10 秒経過
→ OkHttp が「遅いから timeout」

という流れになっている。

Render がコンパイルしている、というより「コンテナ起動中で返事が遅い」 と考えるのが近いです。

2. 対応策：Android 側のタイムアウトを伸ばす
OkHttpClient を Builder で作り直して、
readTimeout を長め（60秒くらい） にしておけば、
Render のコールドスタートを待ち切れるようになります。

import java.util.concurrent.TimeUnit

private val client = OkHttpClient()

// タイムアウトを長めに取った OkHttpClient
private val client: OkHttpClient = OkHttpClient.Builder()
    .connectTimeout(30, TimeUnit.SECONDS)   // 接続まで最大30秒
    .writeTimeout(30, TimeUnit.SECONDS)     // 送信タイムアウト
    .readTimeout(60, TimeUnit.SECONDS)      // レスポンス待ち最大60秒
    .build()
